client--- c_id, consignee_name, s_id, buyer_name, cbm, quantity_incoming, eta

create table client (c_id int PRIMARY KEY, s_id int,consignee_name varchar(255), buyer_name varchar(255), cbm varchar(255), quantity_incoming varchar(255), eta varchar(255),FOREIGN KEY(s_id) REFERENCES supplier s_id  )


create table orders(o_id int PRIMARY KEY, c_id int, quantity_ordered varchar(255), cbm varchar(255), eta varchar(255), FOREIGN KEY (c_id) REFERENCES client(c_id))
